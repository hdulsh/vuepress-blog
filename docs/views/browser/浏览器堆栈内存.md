## 题
```js
/*百度面试题*/
let a = 12;
let b = a;
b = 13;
console.log(a);

let a = {n:12};
let b = a;
b['n'] = 13;
console.log(a.n);

let a = {n:12};
let b = a;
b = {n:13};
console.log(a.n);
```

```js
let a = {
    n: 10
};
let b = a;
b.m = b = {
    n: 20
};
console.log(a);  //{n: 10, m: {n: 20}}
console.log(b);  //{n: 20}
```

```js
let x = [12, 23];
function fn(y) {
    y[0] = 100;
    y = [100];
    y[1] = 200;
    console.log(y);
}
fn(x); 
console.log(x);
//  [100, 200] [100, 23]
```
